\begin{frame}{定义}
	\only<1->{\begin{definition}[素数/质数]
			\label{prime:def:prime}
			只能被 1 和自身整除且大于 1 的数
		\end{definition}}

	\only<2->{\begin{definition}[合数]
			\label{prime:def:nprime}
			不是质数且大于 1 的数
		\end{definition}}

	\only<3->{\begin{alertblock}{注意}
			1 既不是质数也不是合数
		\end{alertblock}}
\end{frame}
