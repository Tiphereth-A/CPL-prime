\begin{frame}[fragile]{素数分布}
	\begin{theorem}
		\label{prime:th:prime-distributation}

		如果令\(\pi(n)\)表示\([1,n]\)中的素数个数, 则我们有
		\footnote{或者写成 \(\displaystyle\lim_{n\to\infty}\frac{\pi(n)}{n (\ln n)^{-1}}=1\)}

		\begin{equation}
			\pi(n)\sim\frac{n}{\ln n}~(n\to\infty)
		\end{equation}
	\end{theorem}
\end{frame}
