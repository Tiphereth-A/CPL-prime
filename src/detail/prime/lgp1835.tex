	\textbf{题目描述} 给定区间 \([L,R]\) (\(1\leq L\leq R < 2^{31}, R-L\leq 10^6\)), 请计算区间中素数的个数

	\textbf{输入格式}

	第一行, 两个正整数 \(L\) 和 \(R\)

	\textbf{输出格式}

	一行, 一个整数, 表示区间中素数的个数

	\textbf{样例输入}

	\includecode[common]{lgp1835.in}

	\textbf{样例输出}

	\includecode[common]{lgp1835.ans}
\end{frame}


\begin{frame}[fragile]{题解}
	筛出 \([1,\sqrt{R}]\) 的质数, 然后用这些质数筛 \([L,R]\) 即可
