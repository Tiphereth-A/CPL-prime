\begin{frame}{时间复杂度}
	\only<1->{易知

		\begin{equation}
			T(n)=\sum_{p\in\{\texttt{prime}\}\cap[2,n]}\left\lfloor\frac{n}{i}\right\rfloor
		\end{equation}}

	\only<2->{而

		\[
			\begin{aligned}
				O(T(n)) & =O\left(\sum_{p\in\{\texttt{prime}\}\cap[2,n]}\left\lfloor\frac{n}{i}\right\rfloor\right) \\
				        & =O\left(n\sum_{p\in\{\texttt{prime}\}\cap[2,n]}\frac{1}{p}\right)                         \\
				        & =O(n\log\log n)
			\end{aligned}
		\]}

	\only<3->{故时间复杂度为\(O(n\log\log n)\)}
\end{frame}
